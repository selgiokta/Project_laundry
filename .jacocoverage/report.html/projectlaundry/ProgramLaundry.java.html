    private void btnOsActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnOsActionPerformed

        int[] rows = tb.getSelectedRows();
        for(int i=0;i<rows.length;i++){
            model.removeRow(rows[i]-i);
        }

    private void btnSImpanActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnSImpanActionPerformed
        
        String Nama = txtNama.getText();
        String Jenis =  cboJenis.getSelectedItem().toString();
        String Kategori = cboKategori.getSelectedItem().toString();
        String Kg = txtKg.getText();
        String Harga = txtHarga.getText();

        Laundry.add(c,Nama);
        c++;
        Laundry.add(c,Harga);
        c++;
        Laundry.add(c,Kategori);
        c++;
        Laundry.add(c,Kg);
        c++;
        Laundry.add(c,Jenis);
        c++;
        
        DefaultTableModel model = (DefaultTableModel) tb.getModel();
        //String kategori = cboKategori.getSelecterdI
        
        int x=startFrom;
        model.addRow(new Object[] {Laundry.get(x), Laundry.get(x+1), Laundry.get(x+2), Laundry.get(x+3), Laundry.get(x+4)});

        startFrom+=5;
        
    }//GEN-LAST:event_btnSImpanActionPerformed

    public int Hitung(int total, int valueKg, int valueCuci){
        total = valueKg * valueCuci;
        txtHarga.setText("Rp." + Integer.toString(total) + ",-");
        return total;
    }
